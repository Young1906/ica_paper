def _plot(sig, **kwargs):
    
    # Sampling frequency
    assert kwargs.get("fs"), "_plot: Missing sample frequency"
    fs = kwargs.get("fs")
    
    # Shape of data:
    n, _l = sig.shape
    t = np.linspace(0, _l/fs, _l)
    
    # Global min and max of data
    _min, _max = np.min(sig), np.max(sig)
    
    # Create figure
    fig = plt.figure()
    
    # MARKER >>>
    
    # first axis is used for marker(s)
    l, b, w, h = 0, 1 - 1/(n+2), 1, 1/(n+1)
    ax0 = fig.add_axes([l,b,w,h])
    ax0.set(xlim=(0, np.max(t)))
    # Get marker
    markers = kwargs.get("markers")
    """
    Marker:
    ["t":1, "c":"black"]
    - t in seconds
    - c plt color pallete
    """
    
    if markers:
        for i in markers:
            _t = i["t"]
            c = i["c"]
            ax0.scatter(_t , 0, marker=11, s=500, color=c)
    
    ax0.axis("off")    

    # <<< MARKER
    
    
    # DATA >>>
    # Get channels / comp names:
    ch_names = kwargs.get("ch_names")
    
    # Function to clean channel_names
    
    def f(t):
        t = str(t)
        t = t.upper()
        return t
    
    ch_names = list(map(f, ch_names))
    
    for i in range(n):
        
        b = b - h
        ax1 = fig.add_axes([l,b,w,h], sharex=ax0, frameon=False)
        
        # option
        ax1.set(ylim=(_min, _max))
        ax1.axis("off")

        # Plotting the data
        ax1.plot(
            t,
            sig[i,:],
            color="black",
            linewidth = 1,)
        
        # Display x-axis if i == n - 1 (last channel)
        ax1.axis(option="tight")
        ax1.text(-.01, .5, ch_names[i], 
                horizontalalignment="right",
                verticalalignment="center",
                fontdict =
                 {
                     'color':  'black',
                     'weight': 'bold',
                     'size': 18,
                 }
                ) 
        
        
    # <<< DATA 
    
    # SCALE >>>
    
    # Temporal scale on last axis
    _len = 1
    _smax = max(t)
    _smin = _smax - _len
    _smid = (_smin + _smax) / 2
    
    ax1.scatter(_smin, _min, marker="|", s=500, color="black", alpha=.25)
    ax1.scatter(_smax, _min, marker="|", s=500, color="black", alpha=.25)
    ax1.hlines(_min, _smin, _smax, alpha=.25)
    
    ax1.text(_smid, _min, "1s",
             horizontalalignment="center",
             verticalalignment="center",
             fontdict =
             {
                 'color':  'black',
                 'size': 18,
             })
    
    # Vertical scale
    l, b, w = 1.025, 0, .1
    ax2 = fig.add_axes([l,b,w,h], sharex=ax0, frameon=False)
    ax2.set(ylim=(_min, _max))
    ax2.axvline(x = 0, linewidth = 5, color="black")
    
    ax2.axis("off")
    _s = np.round(_max - _min, 2)
    
    ax2.text(.1, .5, f" {_s} mV",
             horizontalalignment="left",
             verticalalignment="center",
             fontdict =
             {
                 'color':  'black',
                 'size': 18,
             })
    # <<< SCALE
    return fig

def plot_eeg(sig):
    
    # Create figure
    fig = figure.Figure()
    n, l = sig.shape # n: number of channels / components; l is lenght of the signal
    t = np.linspace(0,l/500, l)
    
    #Number of axes
    n_ = n + 2 # one for each component, first axes is used for annotation, last for scale
    
    for i in range(n_):
        
        l, b, w, h = 0, 1-(i+1)/n_, 1, 1/n_
        _sig = sig[i,:]
        _min, _max = np.min(_sig), np.max(_sig)
        
        # Signal
        ax = fig.add_axes([l, b, w, h], frameon=False)
        ax.plot(t, _sig, color='black', linewidth=.5)
        
        # Channel names
        ax.text(-.1,.5, "Test",verticalalignment='center')
        
        # Other options
        ax.axis('off')
        
        # Scale
        if i == (n-1):
            l, b, w, h = 1, 1-(i+1)/n, 1 / 15, 1/n
            scale = fig.add_axes([l, b, w, h])
            scale.axvline(0, color="black")
#             scale.axhline(0, color="black")
            scale.text(0.1, .5, f"{np.around(_max - _min,2)}", verticalalignment='center')
            scale.axis("off")
    return fig
